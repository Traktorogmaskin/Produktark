      function normalizeWrappedText(value, maxChars, maxLines, maxCharsPerLine) {
        let source = String(value || "").replace(/\r/g, "");
        source = source.slice(0, maxChars);
        const paragraphs = source.split("\n");
        const lines = [];

        for (const paragraph of paragraphs) {
          const words = paragraph.split(/\s+/).filter(Boolean);

          if (!words.length) {
            if (lines.length < maxLines) lines.push("");
            continue;
          }

          let current = "";

          for (const originalWord of words) {
            let word = originalWord;

            while (word.length > maxCharsPerLine) {
              const part = word.slice(0, maxCharsPerLine);
              if (current) {
                if (lines.length < maxLines) lines.push(current);
                current = "";
                if (lines.length >= maxLines) break;
              }
              if (lines.length < maxLines) lines.push(part);
              word = word.slice(maxCharsPerLine);
              if (lines.length >= maxLines) break;
            }

            if (lines.length >= maxLines) break;

            if (!current) {
              current = word;
              continue;
            }

            const candidate = current + " " + word;
            if (candidate.length <= maxCharsPerLine) current = candidate;
            else {
              lines.push(current);
              current = word;
              if (lines.length >= maxLines) break;
            }
          }

          if (lines.length >= maxLines) break;
          if (current) {
            lines.push(current);
            if (lines.length >= maxLines) break;
          }
        }

        let normalized = lines.slice(0, maxLines).join("\n");
        if (normalized.length > maxChars) normalized = normalized.slice(0, maxChars);
        return normalized;
      }

      async function fileToDataUrl(file, maxSize = 1900) {
        return new Promise((resolve, reject) => {
          const reader = new FileReader();
          reader.onload = () => {
            const img = new Image();
            img.onload = () => {
              let width = img.width;
              let height = img.height;

              if (width > height && width > maxSize) {
                height = Math.round((height * maxSize) / width);
                width = maxSize;
              } else if (height >= width && height > maxSize) {
                width = Math.round((width * maxSize) / height);
                height = maxSize;
              }

              const canvas = document.createElement("canvas");
              canvas.width = width;
              canvas.height = height;
              const ctx = canvas.getContext("2d");
              ctx.drawImage(img, 0, 0, width, height);
              resolve(canvas.toDataURL("image/png", 0.92));
            };
            img.onerror = reject;
            img.src = String(reader.result);

        });

      function syncPriceInputs(changedField) {
        if (priceSyncLock) return;
        priceSyncLock = true;

        let list = asNumber(el.listPrice.value);
        let discount = asNumber(el.discountAmount.value);
        let offer = asNumber(el.offerPrice.value);

        const hasList = list > 0;
        const hasDiscount = discount > 0;
        const hasOffer = offer > 0;

        if (hasList && hasDiscount && !hasOffer) {
          offer = Math.max(list - discount, 0);
          el.offerPrice.value = String(offer || "");
        } else if (hasList && hasOffer && !hasDiscount) {
          discount = Math.max(list - offer, 0);
          el.discountAmount.value = String(discount || "");
        } else if (!hasList && hasDiscount && hasOffer) {
          list = discount + offer;
          el.listPrice.value = String(list || "");
        } else if (hasList && hasDiscount && hasOffer) {
          if (changedField === "listPrice") {
            discount = Math.max(list - offer, 0);
            el.discountAmount.value = String(discount || "");
          } else if (changedField === "discountAmount") {
            offer = Math.max(list - discount, 0);
            el.offerPrice.value = String(offer || "");
          } else if (changedField === "offerPrice") {
            discount = Math.max(list - offer, 0);
            el.discountAmount.value = String(discount || "");
          }
        }

        state.listPrice = sanitizeNumber(el.listPrice.value);
        state.discountAmount = sanitizeNumber(el.discountAmount.value);
        state.offerPrice = sanitizeNumber(el.offerPrice.value);
        priceSyncLock = false;
      }

      function resolvePrices() {
        const listPrice = asNumber(state.listPrice);
        const discountAmount = asNumber(state.discountAmount);
        const offerPrice = asNumber(state.offerPrice);

        let activePrice = 0;
        let oldPrice = 0;
        let computedDiscount = 0;
        let showDark = false;
        let activeLabel = "Pris";

        if (listPrice > 0 && offerPrice > 0) {
          activePrice = offerPrice;
          computedDiscount = Math.max(listPrice - offerPrice, 0);
          if (computedDiscount > 0) {
            oldPrice = listPrice;
            showDark = true;
            activeLabel = "Pris etter rabatt";
          }
        } else if (listPrice > 0 && discountAmount > 0) {
          computedDiscount = Math.min(discountAmount, listPrice);
          activePrice = Math.max(listPrice - computedDiscount, 0);
          if (computedDiscount > 0 && activePrice > 0) {
            oldPrice = listPrice;
            showDark = true;
            activeLabel = "Pris etter rabatt";
          } else {
            activePrice = listPrice;
          }
        } else if (listPrice > 0) {
          activePrice = listPrice;
        } else if (offerPrice > 0) {
          activePrice = offerPrice;
        }

        return {
          activePrice,
          oldPrice,
          discountAmount: computedDiscount,
          showDark,
          activeLabel
        };
      }

      function waitForImagesInDocument(doc) {
        const images = Array.from(doc.images || []);
        return Promise.all(
          images.map((img) => {
            if (img.complete && img.naturalWidth > 0) return Promise.resolve();
            return new Promise((resolve) => {
              const done = () => resolve();
              img.addEventListener("load", done, { once: true });
              img.addEventListener("error", done, { once: true });
            });
          })
        );
      }

      async function openPrintWindow() {
        syncStateFromForm();
        renderForm();
        renderPreview();

        const page = buildPageHtml();
        const styles = document.querySelector("style").innerHTML;
        const pdfFilenameBase = buildPdfFilename().replace(/\.pdf$/i, "");

        const printWindow = window.open("", "_blank", "width=1100,height=900");
        if (!printWindow) throw new Error("Popup blocked");

        printWindow.document.open();
        printWindow.document.write(`
          <!DOCTYPE html>
          <html lang="no">
          <head>
            <meta charset="UTF-8" />
            <title>${escapeHtml(pdfFilenameBase)}</title>
            <style>${styles}</style>
          </head>
          <body style="background:#fff;">
            <section class="card preview-card" style="border:none;box-shadow:none;">
              <div class="preview-area" style="padding:0;background:#fff;">
                <div class="preview-pages">${page}</div>
              </div>
            </section>
          </body>
          </html>
        `);
        printWindow.document.close();

        return new Promise((resolve, reject) => {
          const startPrint = async () => {
            try {
              await waitForImagesInDocument(printWindow.document);
              await new Promise((r) => setTimeout(r, 350));

              const closePrintWindow = () => {
                try { printWindow.close(); } catch (e) {}
              };

              if (typeof printWindow.addEventListener === "function") {
                printWindow.addEventListener("afterprint", closePrintWindow, { once: true });
              } else {
                printWindow.onafterprint = closePrintWindow;
              }

              printWindow.focus();
              printWindow.print();
              resolve(printWindow);
            } catch (error) {
              reject(error);
            }
          };

          if (printWindow.document.readyState === "complete") {
            startPrint();
            return;
          }

          printWindow.addEventListener("load", startPrint, { once: true });
        });
      }

        }

          schedulePreview();
        });
      }

      function bindImageUploader() {
        el.btnAddImages.addEventListener("click", () => el.multiImageInput.click());

        el.multiImageInput.addEventListener("change", async (e) => {
          const files = e.target.files;
          if (!files || !files.length) return;
          await addFilesToImages(files);
          e.target.value = "";
        });

        el.btnClearImages.addEventListener("click", () => {
          state.images = [];
          renderInputGallery();
          schedulePreview();
          setStatus("Bilder tømt.");
        });

        ["dragenter", "dragover"].forEach((type) => {
          el.dropzone.addEventListener(type, (e) => {
            e.preventDefault();
            e.stopPropagation();
            el.dropzone.classList.add("dragover");
          });
        });

        ["dragleave", "drop"].forEach((type) => {
          el.dropzone.addEventListener(type, (e) => {
            e.preventDefault();
            e.stopPropagation();
            el.dropzone.classList.remove("dragover");
          });
        });

        el.dropzone.addEventListener("drop", async (e) => {
          const files = e.dataTransfer?.files;
          if (!files?.length) return;
          await addFilesToImages(files);
        });
      }

      function bindLogoInputs() {
        el.btnBrandLogo.addEventListener("click", () => el.brandLogoInput.click());
        el.btnVendorLogo.addEventListener("click", () => el.vendorLogoInput.click());

        el.brandLogoInput.addEventListener("change", async (e) => {
          const file = e.target.files?.[0];
          if (!file) return;
          await setSingleLogo(file, "brand");
          e.target.value = "";
        });

        el.vendorLogoInput.addEventListener("change", async (e) => {
          const file = e.target.files?.[0];
          if (!file) return;
          await setSingleLogo(file, "vendor");
          e.target.value = "";
        });

        el.btnClearBrandLogo.addEventListener("click", () => {
          state.brandLogoDataUrl = "";
          renderLogoPreview(el.brandLogoPreview, "", "");
          schedulePreview();
        });

        el.btnClearVendorLogo.addEventListener("click", () => {
          state.vendorLogoDataUrl = "";
          renderLogoPreview(el.vendorLogoPreview, "", "");
          schedulePreview();
        });
      }

      function bindButtons() {
        el.btnTogglePreview.addEventListener("click", () => {
          state.previewVisible = !state.previewVisible;
          applyPreviewMode();
        });

        el.btnReset.addEventListener("click", () => {
          if (!window.confirm("Tømme hele arket?")) return;
          resetState();
        });

        el.btnSaveProject.addEventListener("click", saveProjectFile);
        el.btnOpenProject.addEventListener("click", () => el.projectFileInput.click());
        el.projectFileInput.addEventListener("change", openProjectFile);

        el.btnPrintPdf.addEventListener("click", async () => {
          try {
            setStatus("Åpner utskrift / lagre PDF ...");
            await openPrintWindow();
            setStatus("Bruk utskriftsdialogen for skriv ut eller lagre som PDF.");
          } catch (error) {
            console.error(error);
            alert("Kunne ikke starte utskrift. Tillat popup-vinduer og prøv igjen.");
            setStatus("Utskriftsfeil.", true);
          }
        });
      }

      function init() {
        renderForm();
        renderSpecTable();
        renderEquipmentTable();
        renderAccessoriesTable();
        renderInputGallery();
        renderLogoPreview(el.brandLogoPreview, "", "");
        renderLogoPreview(el.vendorLogoPreview, "", "");
        applyPreviewMode();
        bindFormEvents();
        bindTables();
        bindImageUploader();
        bindLogoInputs();
        bindButtons();
        renderPreview();
        updateAllCounters();
        setStatus("Klar.");
      }

      init();
    })();
  </script>
